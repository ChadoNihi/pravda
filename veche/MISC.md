## Example start scripts for Alice
```bash
#!/bin/sh

export TC_P2P_PORT=5221
export TC_RPC_PORT=46657
export TC_ABCI_PORT=46658

export TC_API_HOST="localhost"
export TC_API_PORT=8081

export TC_IS_VALIDATOR=true
export TC_DATA=/tmp/timechain/alice
export TC_SEEDS="127.0.0.1:5224"
export TC_GENESIS_TIME=0001-01-01T00:00:00Z
export TC_GENESIS_CHAIN_ID=demonet

export TC_PAYMENT_WALLET_PRIVATE_KEY=31C67548A1A7D662648E4BFDF252E0E232D05FD5DD5144D4CD69F1D50ED8FEBD67EA4654C7F00206215A6B32C736E75A77C0B066D9F5CEDD656714F1A8B64A45
export TC_PAYMENT_WALLET_ADDRESS=67EA4654C7F00206215A6B32C736E75A77C0B066D9F5CEDD656714F1A8B64A45

export BOB_WALLET=17681F651544420EB9C89F055500E61F09374B605AA7B69D98B2DEF74E8789CA
export ALICE_WALLET=67EA4654C7F00206215A6B32C736E75A77C0B066D9F5CEDD656714F1A8B64A45
```
